#!/data/data/com.termux/files/usr/bin/sh
export LANG=zh_TW.UTF-8

# 工具路徑
PYTHON='/data/data/com.termux/files/usr/bin/python3'
CURL='/data/data/com.termux/files/usr/bin/curl'

# GitHub Raw 清單連結
URL_LIST='https://raw.githubusercontent.com/hqpstw-hub/-/6e25c6082d1ed33b9df680bd8f499ecedba53912/twTVlist.txt'
SRC='/data/data/com.termux/files/home/twTVlist.txt'
DEST='/sdcard/Download/twTV.m3u'

echo "🌐 [STEP 1] 下載 GitHub 清單..."
$CURL -L -k -s "$URL_LIST" -o "$SRC"

echo "#EXTM3U" > "$DEST"
echo "🚀 [STEP 2] 解析中..."

while read -r line; do
    [ -z "$line" ] && continue
    LINK=$(echo "$line" | grep -o "https://www.youtube.com/[^ ]*")
    NAME=$(echo "$line" | cut -d',' -f1 | cut -d' ' -f1)

    if [ -n "$LINK" ]; then
        echo "📡 解析: $NAME"
        REAL=$($PYTHON -m yt_dlp --no-warnings --extractor-args "youtube:player-client=ios,web" -g -f b "$LINK" 2>/dev/null)
        if [ -n "$REAL" ]; then
            echo "#EXTINF:-1,$NAME" >> "$DEST"
            echo "$REAL" >> "$DEST"
            echo "   ✅ 成功"
        else
            echo "   ❌ 失敗"
        fi
    fi
done < "$SRC"
echo "✨ 完成！檔案在: $DEST"
