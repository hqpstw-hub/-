cat << 'EOF' > ~/tv/run_tv.sh
#!/bin/bash
# 1. 抓取原始清單 (twTVlist.txt)
curl -L "https://raw.githubusercontent.com/hqpstw-hub/-/refs/heads/main/twTVlist.txt" > ~/tv/raw_list.txt

# 2. 抓取指令腳本 (cmd.txt)，先過濾掉 Windows 換行符號 (\r)，再交給 bash 執行
curl -L "https://raw.githubusercontent.com/hqpstw-hub/-/refs/heads/main/cmd.txt" | sed 's/\r//g' | bash

# 3. 提示完成
echo "--- 雲端指令執行完畢 ---"
EOF
